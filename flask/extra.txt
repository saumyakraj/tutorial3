Virtual env:
python -m venv env
source env/bin/activate
To deactivate the Virtual Env -> deactivate

To generate reqirements.txt -> pip3 freeze > requirements.txt
To install from reqirements.txt -> pip3 install -r requirements.txt

post-request data sample:
{
  "amount" : 700.0,
  "status" : "ordered",
  "items" : {
    "item1" : 10,
    "item2" : 20,
    "item3" : 30
  }
}

Create Table:

sudo su postgres
psql
\l
CREATE DATABASE test
\c test

CREATE TYPE status_enum AS ENUM('ordered', 'paid', 'cancelled', 'completed');

CREATE TABLE IF NOT EXISTS Payments (
	orderId SERIAL NOT NULL PRIMARY KEY, 
    date TIMESTAMP,
  	amount FLOAT,
  	status status_enum,
  	items JSON
);

INSERT INTO Payments(amount, status) 
VALUES(99.99, 'paid');

select * from Payments;
